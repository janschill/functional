# Run time

```haskell
f1 xs = reverse (f2 xs [])

f2 [] ys = ys
f2 (x:xs) [] = f2 xs [x]
f2 (x:xs) ys = f2 xs (f3 x ys)

f3 x [] = [x]
f3 x (y:ys) = if x >= y then x:y:ys else y : (f3 x ys)
```

```haskell
T_f3 (n,0) = O(1)
T_f3 (n,m) = T_f3 (n,m-1) + T_(:) (1, n+m-1)
           = T_f3 (n,m-1) + O(1)
           = T_f3 (n,m-2) + O(1) + O(1)
           = T_f3 (n,0) + O(1) + ... + O(1)
           = O(1) + ... + O(1)
           = O(m)
```

```haskell
T_f2 (0,n) = O(1)
T_f2 (n,0) = T_f2(n-1,1)
T_f2 (n,m) = T_f2(n-1,m+1) + T_f3(1,m)
           = T_f2(n-1,m+1) + O(m)
           = T_f2(n-2,m+2) + O(m) + O(m)
           = T_f2(0,m+n) + O(m) + ... + O(m)
           = O(1) + O(m) + ... + O(m)
           = O(n*m)
           = O(n^2)
```

```haskell
T_f1(n) = T_reverse(n) + T_f2 (n,0)
        = O(n^2) + T_f2 (n,0)
        = O(n^2) + O(n^2)
        = O(2n^2)
        = O(n^2)
```
